# 代理模式 Proxy Pattern

> 为对象提供一个替身，控制对原对象的访问

---

## ES6 Proxy

```js
const user = { name: 'Tom', age: 25 };

const proxy = new Proxy(user, {
  get(target, key) {
    console.log(`读取 ${key}`);
    return key in target ? target[key] : undefined;
  },
  set(target, key, value) {
    console.log(`设置 ${key} = ${value}`);
    target[key] = value;
    return true;
  },
  has(target, key) {
    console.log(`检查 ${key} 是否存在`);
    return key in target;
  },
  deleteProperty(target, key) {
    console.log(`删除 ${key}`);
    delete target[key];
    return true;
  }
});
```

## 前端常见场景

### 1. Vue 3 响应式原理
```js
function reactive(obj) {
  return new Proxy(obj, {
    get(target, key) {
      track(target, key); // 收集依赖
      return Reflect.get(target, key);
    },
    set(target, key, value) {
      const oldValue = target[key];
      const result = Reflect.set(target, key, value);
      if (oldValue !== value) {
        trigger(target, key); // 触发更新
      }
      return result;
    }
  });
}
```

### 2. 数据校验代理
```js
function createValidatedUser(data) {
  return new Proxy(data, {
    set(target, key, value) {
      if (key === 'age' && (!Number.isInteger(value) || value < 0)) {
        throw new Error('年龄必须是正整数');
      }
      if (key === 'email' && !/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value)) {
        throw new Error('邮箱格式错误');
      }
      target[key] = value;
      return true;
    }
  });
}
```

### 3. 缓存代理
```js
function createCachedFn(fn) {
  const cache = new Map();
  return new Proxy(fn, {
    apply(target, thisArg, args) {
      const key = JSON.stringify(args);
      if (cache.has(key)) return cache.get(key);
      const result = target.apply(thisArg, args);
      cache.set(key, result);
      return result;
    }
  });
}

const cachedFetch = createCachedFn(fetchData);
```

### 4. 只读代理
```js
function readonly(obj) {
  return new Proxy(obj, {
    get(target, key) {
      const val = Reflect.get(target, key);
      return typeof val === 'object' ? readonly(val) : val; // 深度只读
    },
    set() {
      console.warn(`禁止修改只读属性`);
      return true;
    },
    deleteProperty() {
      console.warn(`禁止删除只读属性`);
      return true;
    }
  });
}
```

### 5. 图片预加载代理
```js
class ImageLoader {
  constructor(src) { this.src = src; }
  display() { console.log('显示图片:', this.src); }
}

class ProxyImage {
  constructor(src) {
    this.src = src;
    this.realImage = null;
  }
  display() {
    if (!this.realImage) {
      console.log('加载中...');
      this.realImage = new ImageLoader(this.src);
    }
    this.realImage.display();
  }
}
```

## 代理 vs 装饰器

- **代理**：控制访问（权限、缓存、延迟加载）
- **装饰器**：增强功能（日志、埋点、格式化）

## Reflect API

```js
// 推荐在 Proxy 中使用 Reflect 操作目标对象
new Proxy(obj, {
  get(target, key, receiver) {
    return Reflect.get(target, key, receiver);
  },
  set(target, key, value, receiver) {
    return Reflect.set(target, key, value, receiver);
  }
});
```
