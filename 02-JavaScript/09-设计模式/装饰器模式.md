# 装饰器模式 Decorator Pattern

> 在不修改原对象的前提下，动态添加新功能

---

## 函数装饰器

```js
// 给函数添加日志功能
function withLog(fn) {
  return function(...args) {
    console.log(`调用 ${fn.name}，参数:`, args);
    const result = fn.apply(this, args);
    console.log(`返回值:`, result);
    return result;
  };
}

function add(a, b) { return a + b; }
const loggedAdd = withLog(add);
loggedAdd(1, 2);
// 调用 add，参数: [1, 2]
// 返回值: 3
```

## 方法装饰器（ES7 Decorator 提案）

```js
// 需要 Babel 转译
function readonly(target, key, descriptor) {
  descriptor.writable = false;
  return descriptor;
}

class Person {
  @readonly
  name = 'Tom';
}

const p = new Person();
p.name = 'Jerry'; // TypeError
```

## 前端常见场景

### 1. React 高阶组件（HOC）
```js
// withAuth 装饰器：给组件添加权限校验
function withAuth(WrappedComponent) {
  return function AuthenticatedComponent(props) {
    if (!props.isAuthenticated) {
      return <div>请先登录</div>;
    }
    return <WrappedComponent {...props} />;
  };
}

// 使用
const ProtectedPage = withAuth(MyPage);
```

### 2. 防抖/节流装饰
```js
function debounce(fn, delay) {
  let timer;
  return function(...args) {
    clearTimeout(timer);
    timer = setTimeout(() => fn.apply(this, args), delay);
  };
}

const handleSearch = debounce((keyword) => {
  fetch(`/api/search?q=${keyword}`);
}, 300);
```

### 3. 缓存装饰
```js
function memoize(fn) {
  const cache = new Map();
  return function(...args) {
    const key = JSON.stringify(args);
    if (cache.has(key)) return cache.get(key);
    const result = fn.apply(this, args);
    cache.set(key, result);
    return result;
  };
}

const fib = memoize(n => n <= 1 ? n : fib(n - 1) + fib(n - 2));
```

### 4. 权限装饰
```js
function withPermission(requiredRole) {
  return function(handler) {
    return function(...args) {
      if (currentUser.role !== requiredRole) {
        throw new Error('权限不足');
      }
      return handler(...args);
    };
  };
}

const deleteArticle = withPermission('admin')(function(id) {
  // 删除文章逻辑
});
```

## 与代理模式的区别

| 特性 | 装饰器模式 | 代理模式 |
|---|---|---|
| 目的 | 增强功能 | 控制访问 |
| 关注点 | 添加行为 | 延迟/拦截/保护 |
| 关系 | 多个装饰器可叠加 | 通常只有一层代理 |
