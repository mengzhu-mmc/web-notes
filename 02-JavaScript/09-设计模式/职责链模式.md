# 职责链模式 Chain of Responsibility

> 多个处理者依次处理请求，直到某个处理者完成处理

---

## 核心思想

```js
class Handler {
  constructor() { this.next = null; }

  setNext(handler) {
    this.next = handler;
    return handler;
  }

  handle(request) {
    if (this.next) return this.next.handle(request);
    return null;
  }
}

class AuthHandler extends Handler {
  handle(request) {
    if (!request.token) return '未授权';
    return super.handle(request);
  }
}

class LogHandler extends Handler {
  handle(request) {
    console.log('日志:', request);
    return super.handle(request);
  }
}

class BusinessHandler extends Handler {
  handle(request) {
    return `处理业务: ${request.data}`;
  }
}

// 组装链
const auth = new AuthHandler();
const log = new LogHandler();
const biz = new BusinessHandler();
auth.setNext(log).setNext(biz);

auth.handle({ token: 'abc', data: '订单' });
```

## 前端常见场景

### 1. Koa/Express 中间件
```js
// Koa 洋葱模型（职责链变体）
async function compose(middlewares, ctx) {
  function dispatch(i) {
    const fn = middlewares[i];
    if (!fn) return Promise.resolve();
    return fn(ctx, () => dispatch(i + 1));
  }
  return dispatch(0);
}

// 使用
const middlewares = [
  async (ctx, next) => { console.log('1-前'); await next(); console.log('1-后'); },
  async (ctx, next) => { console.log('2-前'); await next(); console.log('2-后'); },
  async (ctx) => { console.log('3-业务'); ctx.body = 'OK'; }
];
```

### 2. 错误处理链
```js
const errorHandlers = [
  (err) => err.status === 401 ? '请先登录' : null,
  (err) => err.status === 403 ? '权限不足' : null,
  (err) => err.status === 404 ? '页面不存在' : null,
  (err) => err.status >= 500 ? '服务器错误' : null,
  (err) => '未知错误' // 兜底
];

function handleError(err) {
  for (const handler of errorHandlers) {
    const result = handler(err);
    if (result) return result;
  }
}
```

### 3. 事件冒泡/捕获
```js
// DOM 事件传播就是职责链
// 捕获阶段: window → document → body → div
// 冒泡阶段: div → body → document → window

div.addEventListener('click', handler, true);  // 捕获
div.addEventListener('click', handler, false); // 冒泡
```

### 4. Promise 链
```js
fetch('/api/user')
  .then(res => res.json())       // 处理者 1：解析 JSON
  .then(data => validate(data))  // 处理者 2：校验数据
  .then(data => render(data))    // 处理者 3：渲染
  .catch(err => handleError(err)); // 错误处理者
```

## 优点
- 解耦：发送者和接收者不需要知道彼此
- 灵活：可动态调整链的顺序和成员
- 开闭原则：新增处理者无需修改现有代码
