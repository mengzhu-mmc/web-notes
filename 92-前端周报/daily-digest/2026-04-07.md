### `using` 关键字（ECMAScript 2026）
> JavaScript 首个资源管理关键字，自动处理资源释放，告别 try...finally 地狱

**浏览器支持**：Chrome 134+ / Edge 134+ / Firefox 141+ / Safari ❌

**代码示例**
```javascript
// 传统方式：手动管理资源
function oldWay() {
  const resource = acquireResource();
  try {
    resource.doSomething();
  } finally {
    resource.dispose(); // 容易忘记
  }
}

// ES2026 方式：自动管理
function newWay() {
  using resource = acquireResource(); // 声明时自动绑定清理
  resource.doSomething();
} // 离开作用域自动调用 [Symbol.dispose]()

// 自定义可释放资源
class DatabaseConnection {
  constructor(url) {
    this.connection = connect(url);
  }
  [Symbol.dispose]() {
    this.connection.close();
  }
}

{
  using db = new DatabaseConnection('localhost');
  db.query('SELECT * FROM users');
} // 自动关闭连接
```

### `Error.isError()` 静态方法（ECMAScript 2026）
> 可靠判断值是否为 Error 对象，解决跨 realm 的 instanceof 问题

**浏览器支持**：Chrome 134+ / Edge 134+ / Firefox 138+ / Safari ❌

**代码示例**
```javascript
// 传统方式的问题
const iframe = document.createElement('iframe');
document.body.appendChild(iframe);
const iframeError = new iframe.contentWindow.Error('test');

iframeError instanceof Error; // false！跨 realm 失效
Error.isError(iframeError);    // true！可靠检测

// 基本用法
Error.isError(new TypeError());  // true
Error.isError(new SyntaxError()); // true
Error.isError({ name: "Error" }); // false
Error.isError(null);             // false
Error.isError("Error");          // false
```

### `Array.fromAsync()` 异步数组创建（ECMAScript 2026）
> 从异步可迭代对象创建数组，简化异步数据处理

**浏览器支持**：Chrome 121+ / Edge 121+ / Firefox 115+ / Safari 16.4+

**代码示例**
```javascript
// 从异步生成器创建数组
async function* asyncGenerator() {
  yield Promise.resolve(1);
  yield Promise.resolve(2);
  yield Promise.resolve(3);
}

async function demo() {
  const arr = await Array.fromAsync(asyncGenerator());
  console.log(arr); // [1, 2, 3]
}

// 从异步可迭代对象创建
async function fetchAllPages(urls) {
  const pages = await Array.fromAsync(
    async function* () {
      for (const url of urls) {
        yield fetch(url).then(r => r.json());
      }
    }()
  );
  return pages;
}
```


### CSS Anchor Positioning（锚点定位）
> 让元素精确"钉"在另一个元素旁边，tooltip 定位噩梦终结者

**浏览器支持**：Chrome ✅ / Firefox 🔄开发中 / Safari 🔄开发中

**代码示例**
```css
/* 定义锚点元素 */
.tooltip-trigger {
  anchor-name: --my-anchor;
}

/* 浮层自动跟随锚点 */
.tooltip {
  position: anchor(--my-anchor);
  inset-area: bottom;  /* 定位在锚点下方 */
  
  /* 可选：自动调整位置防止溢出 */
  position-try-fallbacks: flip-block, flip-inline;
}

/* 精确定位到锚点的边缘 */
.popover {
  position: absolute;
  top: anchor(--my-anchor bottom);
  left: anchor(--my-anchor center);
  transform: translateX(-50%);
}
```

**实际效果**：无需 JavaScript，tooltip 自动跟随按钮，页面滚动时保持相对位置。
