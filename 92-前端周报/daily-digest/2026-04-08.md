### `Promise.try()` 静态方法（ECMAScript 2026）
> 统一处理同步/异步函数，无需手动包装 Promise

**浏览器支持**：Chrome 134+ / Edge 134+ / Firefox 141+ / Safari ❌

**代码示例**
```javascript
// 传统方式：需要手动包装
function oldWay() {
  return Promise.resolve().then(() => {
    return syncFunction(); // 同步函数
  });
}

// Promise.try 方式：统一处理
function newWay() {
  return Promise.try(syncFunction); // 同步或异步都行
}

// 实际应用：统一错误处理
async function fetchUserData(userId) {
  return Promise.try(() => {
    if (!userId) throw new Error('userId required');
    return fetch(`/api/users/${userId}`);
  })
  .then(res => res.json())
  .catch(err => {
    console.error('获取用户失败:', err);
    return null;
  });
}

// 支持传参
Promise.try(multiply, 3, 4); // multiply(3, 4)

function multiply(a, b) {
  return a * b;
}
```

### Temporal API（ECMAScript 2026 Stage 3）
> JavaScript 日期处理的革命性升级，告别 Date 对象的坑

**浏览器支持**：Chrome 🔄开发中 / Firefox 🔄开发中 / Safari 🔄开发中（可通过 polyfill 使用）

**代码示例**
```javascript
// 传统 Date 的坑
const date = new Date(2026, 3, 8); // 月份从 0 开始！实际是 4 月 8 日
date.getMonth(); // 返回 3，容易混淆

// Temporal API：直观且不可变
const today = Temporal.Now.plainDateISO();
console.log(today.toString()); // "2026-04-08"

// 创建特定日期（月份从 1 开始！）
const birthday = Temporal.PlainDate.from('2026-12-25');
const newYear = Temporal.PlainDate.from({ year: 2026, month: 1, day: 1 });

// 日期计算
const tomorrow = today.add({ days: 1 });
const nextWeek = today.add({ weeks: 1 });
const age = birthday.since(today, { largestUnit: 'years' });

// 时区处理
const zoned = Temporal.Now.zonedDateTimeISO('Asia/Shanghai');
const inNewYork = zoned.withTimeZone('America/New_York');

// 时间段计算
const duration = Temporal.Duration.from({ hours: 2, minutes: 30 });
const later = Temporal.Now.plainTimeISO().add(duration);
```

**核心优势**
- 不可变对象，避免意外修改
- 月份从 1 开始，符合直觉
- 时区处理标准化
- 精确到纳秒级别

### React 19 `useActionState` Hook
> 简化异步操作的状态管理，自动处理 pending/error 状态

**浏览器支持**：React 19+（所有现代浏览器）

**代码示例**
```javascript
import { useActionState } from 'react';

// 传统方式：手动管理状态
function OldForm() {
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState(null);
  
  const handleSubmit = async (formData) => {
    setLoading(true);
    setError(null);
    try {
      await submitForm(formData);
    } catch (err) {
      setError(err.message);
    } finally {
      setLoading(false);
    }
  };
  
  return (
    <form onSubmit={handleSubmit}>
      <button disabled={loading}>
        {loading ? '提交中...' : '提交'}
      </button>
      {error && <p className="error">{error}</p>}
    </form>
  );
}

// useActionState 方式：自动管理
function NewForm() {
  const [error, submitAction, isPending] = useActionState(
    async (previousState, formData) => {
      const result = await submitForm(formData);
      if (result.error) return result.error;
      return null; // 成功返回 null
    },
    null // 初始状态
  );
  
  return (
    <form action={submitAction}>
      <button type="submit" disabled={isPending}>
        {isPending ? '提交中...' : '提交'}
      </button>
      {error && <p className="error">{error}</p>}
    </form>
  );
}
```

### React 19 `useOptimistic` Hook
> 实现乐观更新，提升用户体验

**浏览器支持**：React 19+（所有现代浏览器）

**代码示例**
```javascript
import { useOptimistic, useState } from 'react';

function TodoList({ todos, updateTodo }) {
  const [optimisticTodos, setOptimisticTodo] = useOptimistic(
    todos,
    (currentTodos, newTodo) => {
      // 立即显示新状态（乐观更新）
      return currentTodos.map(t => 
        t.id === newTodo.id ? { ...t, ...newTodo } : t
      );
    }
  );
  
  async function handleToggle(todo) {
    // 立即更新 UI
    setOptimisticTodo({ id: todo.id, completed: !todo.completed });
    
    // 发送请求
    await updateTodo(todo.id, { completed: !todo.completed });
  }
  
  return (
    <ul>
      {optimisticTodos.map(todo => (
        <li key={todo.id}>
          <input 
            type="checkbox" 
            checked={todo.completed}
            onChange={() => handleToggle(todo)}
          />
          {todo.text}
        </li>
      ))}
    </ul>
  );
}
```

### CSS Anchor Positioning（锚点定位）
> 纯 CSS 实现 tooltip/下拉菜单定位，自动处理碰撞检测

**浏览器支持**：Chrome 125+ / Edge 125+ / Firefox 🔄开发中 / Safari 🔄开发中

**代码示例**
```css
/* 定义锚点 */
.trigger-button {
  anchor-name: --my-anchor;
}

/* tooltip 自动跟随锚点 */
.tooltip {
  position: absolute;
  position-anchor: --my-anchor;
  inset-area: top;  /* 定位在锚点上方 */
  margin-bottom: 8px;
}

/* 自动回退：上方放不下就放下方 */
.tooltip {
  position-try-fallbacks: flip-block, flip-inline;
}

/* 实际应用：下拉菜单 */
.dropdown-trigger {
  anchor-name: --dropdown-anchor;
}

.dropdown-menu {
  position: absolute;
  position-anchor: --dropdown-anchor;
  top: anchor(--dropdown-anchor bottom);
  left: anchor(--dropdown-anchor left);
}
```

**实际效果**
- 下拉菜单自动贴着触发按钮
- 视口边缘自动翻转位置
- 无需 JavaScript 计算

### CSS `calc-size()` 函数
> 支持动画到 `auto` 尺寸，实现平滑展开/收起动画

**浏览器支持**：Chrome 129+ / Edge 129+ / Firefox 🔄开发中 / Safari 🔄开发中

**代码示例**
```css
/* 传统方式：无法动画到 auto */
.content {
  height: 0;
  overflow: hidden;
  transition: height 0.3s;
}
.content.expanded {
  height: auto; /* 无法动画！ */
}

/* calc-size 方式：平滑动画 */
.content {
  height: 0;
  overflow: hidden;
  transition: height 0.3s;
}

.content.expanded {
  height: calc-size(auto, size); /* 可动画！ */
}

/* 实际应用：手风琴组件 */
.accordion-item {
  --max-height: calc-size(auto, size);
  max-height: 0;
  overflow: hidden;
  transition: max-height 0.3s ease-out;
}

.accordion-item.open {
  max-height: var(--max-height);
}
```
