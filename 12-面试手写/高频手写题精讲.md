## 一、手写 instanceof

```javascript
function myInstanceof(left, right) {
  let proto = Object.getPrototypeOf(left);
  const prototype = right.prototype;
  while (proto !== null) {
    if (proto === prototype) return true;
    proto = Object.getPrototypeOf(proto);
  }
  return false;
}

// 测试
myInstanceof([], Array);   // true
myInstanceof([], Object);  // true
myInstanceof([], Map);     // false
```

**原理**：沿着左侧的原型链向上找，看能否找到右侧的 `prototype`。

## 二、数组扁平化

```javascript
// 方法一：递归（支持指定深度）
function flatten(arr, depth = Infinity) {
  return arr.reduce((acc, val) => {
    if (Array.isArray(val) && depth > 0) {
      return acc.concat(flatten(val, depth - 1));
    }
    return acc.concat(val);
  }, []);
}

// 方法二：迭代（用栈模拟，完全扁平化）
function flattenIterative(arr) {
  const stack = [...arr];
  const result = [];
  while (stack.length) {
    const item = stack.pop();
    if (Array.isArray(item)) {
      stack.push(...item);
    } else {
      result.unshift(item);
    }
  }
  return result;
}

// 方法三：原生 API
[1, [2, [3]]].flat(Infinity); // [1, 2, 3]
```

## 三、数组去重

```javascript
// 方法一：Set（最简洁）
const unique = arr => [...new Set(arr)];

// 方法二：filter + indexOf
const unique2 = arr => arr.filter((item, index) => arr.indexOf(item) === index);

// 方法三：reduce
const unique3 = arr => arr.reduce((acc, item) =>
  acc.includes(item) ? acc : [...acc, item], []);

// 方法四：按 key 去重（处理对象数组）
function uniqueBy(arr, key) {
  const map = new Map();
  return arr.filter(item => {
    const k = typeof key === 'function' ? key(item) : item[key];
    if (map.has(k)) return false;
    map.set(k, true);
    return true;
  });
}

// 使用
uniqueBy([{id: 1}, {id: 2}, {id: 1}], 'id'); // [{id:1}, {id:2}]
```

## 四、并发控制调度器

```javascript
// 带并发限制的异步调度器
class Scheduler {
  constructor(maxConcurrent) {
    this.maxConcurrent = maxConcurrent;
    this.running = 0;
    this.queue = [];
  }

  add(promiseCreator) {
    return new Promise((resolve, reject) => {
      this.queue.push(() => promiseCreator().then(resolve, reject));
      this.run();
    });
  }

  run() {
    while (this.running < this.maxConcurrent && this.queue.length) {
      const task = this.queue.shift();
      this.running++;
      task().finally(() => {
        this.running--;
        this.run();
      });
    }
  }
}

// 使用示例
const scheduler = new Scheduler(2); // 最多同时执行 2 个
const delay = (ms) => new Promise(resolve => setTimeout(resolve, ms));

scheduler.add(() => delay(1000).then(() => console.log('1')));
scheduler.add(() => delay(500).then(() => console.log('2')));
scheduler.add(() => delay(300).then(() => console.log('3')));
scheduler.add(() => delay(400).then(() => console.log('4')));
// 输出顺序：2 → 3 → 1 → 4
```

**核心思路**：维护一个 running 计数器和等待队列，每完成一个任务就从队列里取下一个执行。

## 五、柯里化

```javascript
function curry(fn) {
  return function curried(...args) {
    // 参数够了就执行，不够就继续收集
    if (args.length >= fn.length) {
      return fn.apply(this, args);
    }
    return function (...moreArgs) {
      return curried.apply(this, [...args, ...moreArgs]);
    };
  };
}

// 使用
const add = (a, b, c) => a + b + c;
const curriedAdd = curry(add);
curriedAdd(1)(2)(3);    // 6
curriedAdd(1, 2)(3);    // 6
curriedAdd(1)(2, 3);    // 6
```

**核心思路**：判断已收集参数数量是否达到函数形参数量，未达到就返回新函数继续收集。

## 六、函数组合（compose / pipe）

```javascript
// compose：从右到左执行
const compose = (...fns) => x => fns.reduceRight((acc, fn) => fn(acc), x);

// pipe：从左到右执行
const pipe = (...fns) => x => fns.reduce((acc, fn) => fn(acc), x);

// 使用
const add1 = x => x + 1;
const double = x => x * 2;
const square = x => x * x;

compose(square, double, add1)(3); // square(double(add1(3))) = square(double(4)) = square(8) = 64
pipe(add1, double, square)(3);    // 同上，从左到右，结果一样
```
