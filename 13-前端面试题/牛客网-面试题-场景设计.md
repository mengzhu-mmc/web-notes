## 1. 实现一个虚拟列表

**难度**：⭐⭐⭐⭐⭐ | **频率**：🔥🔥🔥🔥🔥

**思路：**
- 只渲染**可视区域**内的 DOM 节点（通常 10-20 个）
- 用一个高度等于总数据高度的占位 div 撑开滚动条
- 监听 scroll 事件，计算当前应渲染的起止索引
- 用 `transform: translateY` 将可见区域偏移到正确位置

```jsx
import { useState, useRef, useCallback } from 'react';

function VirtualList({ items, itemHeight = 50, containerHeight = 400, overscan = 3 }) {
  const [scrollTop, setScrollTop] = useState(0);

  const visibleCount = Math.ceil(containerHeight / itemHeight);
  const startIndex = Math.max(0, Math.floor(scrollTop / itemHeight) - overscan);
  const endIndex = Math.min(items.length - 1, startIndex + visibleCount + overscan * 2);

  const visibleItems = items.slice(startIndex, endIndex + 1);
  const totalHeight = items.length * itemHeight;
  const offsetY = startIndex * itemHeight;

  const onScroll = useCallback((e) => {
    setScrollTop(e.currentTarget.scrollTop);
  }, []);

  return (
    <div
      style={{ height: containerHeight, overflowY: 'auto', position: 'relative' }}
      onScroll={onScroll}
    >
      <div style={{ height: totalHeight }}>
        <div style={{ transform: `translateY(${offsetY}px)` }}>
          {visibleItems.map((item, i) => (
            <div
              key={startIndex + i}
              style={{
                height: itemHeight,
                display: 'flex',
                alignItems: 'center',
                padding: '0 16px',
                borderBottom: '1px solid #eee',
              }}
            >
              {item.text}
            </div>
          ))}
        </div>
      </div>
    </div>
  );
}

// 测试
const data = Array.from({ length: 100000 }, (_, i) => ({ id: i, text: `Item #${i}` }));
export default () => <VirtualList items={data} itemHeight={50} containerHeight={400} />;
```

## 2. 实现大文件上传（分片 + 断点续传）

**难度**：⭐⭐⭐⭐⭐ | **频率**：🔥🔥🔥🔥

**思路：**
1. **分片**：将文件按固定大小（如 2MB）切割成多个 chunk
2. **计算 hash**：用 `spark-md5` 计算文件 hash，作为唯一标识
3. **询问服务器**：上传前检查哪些分片已上传（断点续传）
4. **并发上传**：并发上传未上传的分片（控制并发数）
5. **合并**：所有分片上传完毕后，通知服务器合并

```js
import SparkMD5 from 'spark-md5';

const CHUNK_SIZE = 2 * 1024 * 1024; // 2MB

// 1. 切片
function createChunks(file, chunkSize = CHUNK_SIZE) {
  const chunks = [];
  let start = 0;
  while (start < file.size) {
    chunks.push(file.slice(start, start + chunkSize));
    start += chunkSize;
  }
  return chunks;
}

// 2. 计算文件 hash（使用 Web Worker 避免阻塞主线程）
function calcFileHash(chunks) {
  return new Promise((resolve) => {
    const spark = new SparkMD5.ArrayBuffer();
    let index = 0;

    function loadChunk() {
      if (index >= chunks.length) {
        resolve(spark.end());
        return;
      }
      const reader = new FileReader();
      reader.onload = (e) => {
        spark.append(e.target.result);
        index++;
        loadChunk();
      };
      reader.readAsArrayBuffer(chunks[index]);
    }
    loadChunk();
  });
}

// 3. 控制并发上传
async function uploadWithConcurrency(tasks, limit = 3) {
  const results = [];
  const pool = new Set();

  for (const task of tasks) {
    const p = task().then(res => { pool.delete(p); return res; });
    pool.add(p);
    results.push(p);
    if (pool.size >= limit) {
      await Promise.race(pool);
    }
  }
  return Promise.all(results);
}

// 4. 主上传函数
async function uploadFile(file) {
  const chunks = createChunks(file);
  const fileHash = await calcFileHash(chunks);

  // 询问服务器：哪些分片已上传
  const { uploadedChunks } = await fetch('/api/upload/check', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ fileHash, filename: file.name }),
  }).then(r => r.json());

  const uploadedSet = new Set(uploadedChunks);

  // 过滤已上传的分片
  const tasks = chunks
    .filter((_, index) => !uploadedSet.has(index))
    .map((chunk, index) => () => {
      const formData = new FormData();
      formData.append('chunk', chunk);
      formData.append('fileHash', fileHash);
      formData.append('index', index);
      formData.append('total', chunks.length);
      return fetch('/api/upload/chunk', { method: 'POST', body: formData });
    });

  // 并发上传（最多3个）
  await uploadWithConcurrency(tasks, 3);

  // 通知服务器合并
  await fetch('/api/upload/merge', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ fileHash, filename: file.name, chunkSize: CHUNK_SIZE }),
  });

  console.log('Upload complete!');
}
```

## 3. 实现一个简易 Promise

**难度**：⭐⭐⭐⭐⭐ | **频率**：🔥🔥🔥🔥🔥

**思路：** 状态机（pending/fulfilled/rejected）+ 微任务队列（`queueMicrotask`）+ 链式调用

```js
const PENDING = 'pending';
const FULFILLED = 'fulfilled';
const REJECTED = 'rejected';

class MyPromise {
  #state = PENDING;
  #value = undefined;
  #callbacks = [];  // { onFulfilled, onRejected, resolve, reject }

  constructor(executor) {
    const resolve = (value) => {
      if (this.#state !== PENDING) return;
      // 如果 resolve 的值是 Promise，需要等待它
      if (value instanceof MyPromise) {
        value.then(resolve, reject);
        return;
      }
      this.#state = FULFILLED;
      this.#value = value;
      this.#flush();
    };
    const reject = (reason) => {
      if (this.#state !== PENDING) return;
      this.#state = REJECTED;
      this.#value = reason;
      this.#flush();
    };
    try {
      executor(resolve, reject);
    } catch (e) {
      reject(e);
    }
  }

  #flush() {
    this.#callbacks.forEach(({ onFulfilled, onRejected, resolve, reject }) => {
      queueMicrotask(() => {
        const handler = this.#state === FULFILLED ? onFulfilled : onRejected;
        if (typeof handler !== 'function') {
          this.#state === FULFILLED ? resolve(this.#value) : reject(this.#value);
          return;
        }
        try {
          resolve(handler(this.#value));
        } catch (e) {
          reject(e);
        }
      });
    });
    this.#callbacks = [];
  }

  then(onFulfilled, onRejected) {
    return new MyPromise((resolve, reject) => {
      if (this.#state === PENDING) {
        this.#callbacks.push({ onFulfilled, onRejected, resolve, reject });
      } else {
        queueMicrotask(() => {
          const handler = this.#state === FULFILLED ? onFulfilled : onRejected;
          if (typeof handler !== 'function') {
            this.#state === FULFILLED ? resolve(this.#value) : reject(this.#value);
            return;
          }
          try { resolve(handler(this.#value)); }
          catch (e) { reject(e); }
        });
      }
    });
  }

  catch(onRejected) { return this.then(null, onRejected); }

  finally(onFinally) {
    return this.then(
      value => MyPromise.resolve(onFinally()).then(() => value),
      reason => MyPromise.resolve(onFinally()).then(() => { throw reason; })
    );
  }

  static resolve(value) {
    if (value instanceof MyPromise) return value;
    return new MyPromise(resolve => resolve(value));
  }

  static reject(reason) { return new MyPromise((_, reject) => reject(reason)); }

  static all(promises) {
    return new MyPromise((resolve, reject) => {
      const results = [];
      let count = 0;
      if (!promises.length) return resolve(results);
      promises.forEach((p, i) => {
        MyPromise.resolve(p).then(val => {
          results[i] = val;
          if (++count === promises.length) resolve(results);
        }, reject);
      });
    });
  }
}

// 测试
new MyPromise((resolve) => setTimeout(() => resolve(1), 100))
  .then(v => v + 1)
  .then(v => { console.log(v); return MyPromise.reject('error'); })
  .catch(e => console.log('caught:', e));
```

## 4. 实现一个事件总线（EventEmitter）

**难度**：⭐⭐⭐ | **频率**：🔥🔥🔥🔥

```js
class EventEmitter {
  #events = new Map();

  on(event, listener) {
    if (!this.#events.has(event)) {
      this.#events.set(event, []);
    }
    this.#events.get(event).push(listener);
    return this;  // 支持链式调用
  }

  once(event, listener) {
    const wrapper = (...args) => {
      listener(...args);
      this.off(event, wrapper);
    };
    wrapper._origin = listener;  // 保存原始函数引用，用于 off
    return this.on(event, wrapper);
  }

  off(event, listener) {
    const listeners = this.#events.get(event);
    if (!listeners) return this;
    this.#events.set(
      event,
      listeners.filter(l => l !== listener && l._origin !== listener)
    );
    return this;
  }

  emit(event, ...args) {
    const listeners = this.#events.get(event);
    if (!listeners) return false;
    listeners.slice().forEach(l => l(...args));  // slice 防止 emit 中 off 导致的问题
    return true;
  }

  removeAllListeners(event) {
    if (event) {
      this.#events.delete(event);
    } else {
      this.#events.clear();
    }
    return this;
  }

  listenerCount(event) {
    return this.#events.get(event)?.length ?? 0;
  }
}

// 测试
const emitter = new EventEmitter();

const handler = (msg) => console.log('on:', msg);
emitter.on('message', handler);
emitter.once('message', (msg) => console.log('once:', msg));

emitter.emit('message', 'hello');  // on: hello, once: hello
emitter.emit('message', 'world');  // on: world（once 已被移除）

emitter.off('message', handler);
emitter.emit('message', 'test');   // 无输出
```

## 5. 实现 LRU 缓存

**难度**：⭐⭐⭐⭐ | **频率**：🔥🔥🔥🔥🔥

**思路：** 使用 `Map`（ES6 保证插入顺序）实现 O(1) 的 get/put

```js
class LRUCache {
  #capacity;
  #cache;  // Map 保证顺序，最近使用的在末尾

  constructor(capacity) {
    this.#capacity = capacity;
    this.#cache = new Map();
  }

  get(key) {
    if (!this.#cache.has(key)) return -1;
    // 访问后移到末尾（最近使用）
    const value = this.#cache.get(key);
    this.#cache.delete(key);
    this.#cache.set(key, value);
    return value;
  }

  put(key, value) {
    if (this.#cache.has(key)) {
      this.#cache.delete(key);  // 先删再设置，更新顺序
    } else if (this.#cache.size >= this.#capacity) {
      // 删除最久未使用（Map 第一个元素）
      const oldestKey = this.#cache.keys().next().value;
      this.#cache.delete(oldestKey);
    }
    this.#cache.set(key, value);
  }

  get size() { return this.#cache.size; }
}

// 测试
const cache = new LRUCache(3);
cache.put(1, 'a');   // {1:'a'}
cache.put(2, 'b');   // {1:'a', 2:'b'}
cache.put(3, 'c');   // {1:'a', 2:'b', 3:'c'}
cache.get(1);         // 'a'，访问1后: {2:'b', 3:'c', 1:'a'}
cache.put(4, 'd');   // 容量满，删除最久未用(2): {3:'c', 1:'a', 4:'d'}
console.log(cache.get(2));  // -1（已被淘汰）
console.log(cache.get(3));  // 'c'
```

## 6. 实现并发请求控制（Promise 并发池）

**难度**：⭐⭐⭐⭐ | **频率**：🔥🔥🔥🔥🔥

```js
/**
 * 并发控制：最多同时执行 limit 个异步任务
 * @param {Function[]} tasks - 任务数组，每个任务是返回 Promise 的函数
 * @param {number} limit - 最大并发数
 */
async function concurrentLimit(tasks, limit) {
  const results = new Array(tasks.length);
  const executing = new Set();
  let index = 0;

  async function runNext() {
    if (index >= tasks.length) return;

    const currentIndex = index++;
    const task = tasks[currentIndex];

    const promise = task().then(result => {
      results[currentIndex] = { status: 'fulfilled', value: result };
      executing.delete(promise);
    }).catch(error => {
      results[currentIndex] = { status: 'rejected', reason: error };
      executing.delete(promise);
    });

    executing.add(promise);

    if (executing.size >= limit) {
      await Promise.race(executing);
    }
    await runNext();
  }

  // 初始启动 limit 个任务
  await Promise.all(Array.from({ length: Math.min(limit, tasks.length) }, runNext));
  return results;
}

// 更简洁的版本（保持顺序输出）
async function pLimit(tasks, limit) {
  const results = [];
  const queue = tasks.map((task, i) => ({ task, i }));
  const pool = new Set();

  async function runTask({ task, i }) {
    const p = task().then(v => results[i] = v);
    pool.add(p);
    p.finally(() => pool.delete(p));
    if (pool.size >= limit) await Promise.race(pool);
  }

  for (const item of queue) {
    await runTask(item);
  }
  await Promise.all(pool);
  return results;
}

// 测试
const tasks = Array.from({ length: 10 }, (_, i) => () =>
  new Promise(resolve => setTimeout(() => {
    console.log(`Task ${i} done`);
    resolve(i);
  }, Math.random() * 1000))
);

concurrentLimit(tasks, 3).then(console.log);
// 同时只有 3 个在执行，全部完成后输出结果数组
```

## 7. 实现 JSON.stringify

**难度**：⭐⭐⭐⭐ | **频率**：🔥🔥🔥

```js
function myJSONStringify(value, replacer, space) {
  const indent = typeof space === 'number' ? ' '.repeat(space) : (space || '');

  function stringify(val, currentIndent) {
    // 特殊值：返回 undefined（会被过滤掉）
    if (val === undefined || typeof val === 'function' || typeof val === 'symbol') {
      return undefined;
    }
    if (val === null) return 'null';

    // 有 toJSON 方法，调用它
    if (typeof val.toJSON === 'function') return stringify(val.toJSON());

    if (typeof val === 'boolean') return String(val);
    if (typeof val === 'number') {
      return isFinite(val) ? String(val) : 'null';  // Infinity/NaN → null
    }
    if (typeof val === 'string') {
      return `"${val.replace(/\\/g, '\\\\').replace(/"/g, '\\"').replace(/\n/g, '\\n').replace(/\r/g, '\\r').replace(/\t/g, '\\t')}"`;
    }
    if (typeof val === 'bigint') throw new TypeError('Do not know how to serialize BigInt');

    if (Array.isArray(val)) {
      const nextIndent = currentIndent + indent;
      const items = val.map(item => {
        const result = stringify(item, nextIndent);
        return result === undefined ? 'null' : result;  // 数组中 undefined 变 null
      });
      if (!indent) return `[${items.join(',')}]`;
      return `[\n${items.map(i => nextIndent + i).join(',\n')}\n${currentIndent}]`;
    }

    if (typeof val === 'object') {
      const nextIndent = currentIndent + indent;
      const keys = Object.keys(val);
      const pairs = keys
        .map(key => {
          const v = stringify(val[key], nextIndent);
          if (v === undefined) return undefined;
          const keyStr = stringify(key);
          return indent
            ? `${nextIndent}${keyStr}: ${v}`
            : `${keyStr}:${v}`;
        })
        .filter(p => p !== undefined);

      if (!indent) return `{${pairs.join(',')}}`;
      return `{\n${pairs.join(',\n')}\n${currentIndent}}`;
    }
  }

  return stringify(value, '');
}

// 测试
console.log(myJSONStringify({ a: 1, b: [1, 2, null], c: undefined }));
// '{"a":1,"b":[1,2,null]}'

console.log(myJSONStringify({ a: 1 }, null, 2));
// {
//   "a": 1
// }
```

## 8. 实现 compose / pipe

```js
// compose：从右到左执行（数学函数组合）
// compose(f, g, h)(x) = f(g(h(x)))
const compose = (...fns) => (x) =>
  fns.reduceRight((acc, fn) => fn(acc), x);

// pipe：从左到右执行（更符合直觉）
// pipe(f, g, h)(x) = h(g(f(x)))
const pipe = (...fns) => (x) =>
  fns.reduce((acc, fn) => fn(acc), x);

// 支持异步的 pipeAsync
const pipeAsync = (...fns) => (x) =>
  fns.reduce((acc, fn) => Promise.resolve(acc).then(fn), Promise.resolve(x));
```

## 9. 实现 useState（React Hooks 原理）

**难度**：⭐⭐⭐⭐⭐ | **频率**：🔥🔥🔥🔥

**思路：** 用数组按 Hook 调用顺序存储状态（这就是为什么 Hooks 不能在条件语句中调用）

```js
let hooks = [];      // 存储所有 hook 的状态
let currentIndex = 0; // 当前正在处理的 hook 索引

// 模拟 React 重新渲染
function render(component) {
  currentIndex = 0;       // 每次渲染重置索引
  component();            // 执行组件函数（会依次调用 hooks）
}

function useState(initialState) {
  const index = currentIndex++;  // 记录当前 hook 的索引（闭包捕获）

  // 首次初始化
  if (hooks[index] === undefined) {
    hooks[index] = typeof initialState === 'function'
      ? initialState()
      : initialState;
  }

  const setState = (newState) => {
    const next = typeof newState === 'function'
      ? newState(hooks[index])
      : newState;

    if (Object.is(next, hooks[index])) return;  // 相同则不更新

    hooks[index] = next;
    render(App);  // 触发重新渲染（实际 React 是调度更新）
  };

  return [hooks[index], setState];
}

function useEffect(callback, deps) {
  const index = currentIndex++;
  const hook = hooks[index] || {};

  const hasChanged = !hook.deps || deps.some((d, i) => !Object.is(d, hook.deps[i]));

  if (hasChanged) {
    if (hook.cleanup) hook.cleanup();  // 执行上次的清理函数
    // 异步执行（实际在 commit 阶段后）
    Promise.resolve().then(() => {
      hook.cleanup = callback();
    });
  }

  hooks[index] = { deps };
}

// 测试
function App() {
  const [count, setCount] = useState(0);
  const [text, setText] = useState('hello');
  console.log('render:', count, text);
}

render(App);  // render: 0 'hello'
// hooks[0] 的 setState 被调用：setCount(1)
hooks[0] = 1; currentIndex = 0;
render(App);  // render: 1 'hello'
```

## 10. 实现 useEffect

**难度**：⭐⭐⭐⭐ | **频率**：🔥🔥🔥🔥

（见上方 useState 中的 useEffect 实现，这里展示独立的完整版本）

```js
// 完整的 useEffect 实现要点：
// 1. 比较 deps：用 Object.is 逐项比较
// 2. 首次渲染：deps 为 undefined，视为已变化
// 3. 清理函数：下次执行 effect 前或卸载时调用
// 4. 执行时机：浏览器绘制后（异步）

const effectHooks = [];
let effectIndex = 0;

function useEffect(callback, deps) {
  const index = effectIndex++;
  const prev = effectHooks[index];

  // 判断是否需要执行
  const shouldRun = !prev ||  // 首次
    !deps ||  // 无 deps，每次都执行
    deps.length !== prev.deps?.length ||
    deps.some((dep, i) => !Object.is(dep, prev.deps[i]));

  if (shouldRun) {
    // 模拟异步（实际在 commit 后 layout effect 前 / 后）
    queueMicrotask(() => {
      // 执行清理函数
      if (prev?.cleanup) prev.cleanup();
      // 执行 effect，保存清理函数
      const cleanup = callback();
      effectHooks[index] = { deps, cleanup };
    });
  } else {
    effectHooks[index] = { ...prev };
  }
}
```

class Analytics {
  #config;
  #queue = [];
  #timer = null;

  constructor(config = {}) {
    this.#config = {
      endpoint: '/api/track',
      batchSize: 10,      // 队列满 10 条时发送
      flushInterval: 5000, // 5秒强制发送
      appId: 'default',
      ...config,
    };
    this.#startAutoFlush();
    this.#initAutoTracking();
  }

  // 核心：加入队列
  track(eventName, properties = {}) {
    this.#queue.push({
      event: eventName,
      properties: {
        ...properties,
        appId: this.#config.appId,
        url: location.href,
        timestamp: Date.now(),
        userAgent: navigator.userAgent,
        screenWidth: screen.width,
        screenHeight: screen.height,
      },
    });

    if (this.#queue.length >= this.#config.batchSize) {
      this.#flush();
    }
  }

  // 发送数据

    const payload = JSON.stringify(batch);

    if (navigator.sendBeacon) {
      navigator.sendBeacon(this.#config.endpoint, payload);
    } else {
      fetch(this.#config.endpoint, {
        method: 'POST',
        body: payload,
        headers: { 'Content-Type': 'application/json' },
        keepalive: true,  // 页面关闭后也能发送
      }).catch(() => {});
    }
  }

  // 定时发送
  #startAutoFlush() {
    this.#timer = setInterval(() => this.#flush(), this.#config.flushInterval);
    // 页面卸载时发送剩余数据
    window.addEventListener('beforeunload', () => this.#flush());
    document.addEventListener('visibilitychange', () => {
      if (document.visibilityState === 'hidden') this.#flush();
    });
  }

  // 自动追踪
  #initAutoTracking() {

    this.track('page_view');

    // 单页应用 SPA 路由变化

    // 全局错误

    });
  }

  identify(userId, traits = {}) {

  }

  destroy() {

```js
  thisArg = thisArg == null ? globalThis : Object(thisArg);
  const sym = Symbol('fn');
  thisArg[sym] = this;
  const result = thisArg[sym](...args);
  delete thisArg[sym];

Function.prototype.myBind = function(thisArg, ...preArgs) {
  const fn = this;
  function bound(...args) {
    // new 调用时，this 是新实例，不用 thisArg
    if (new.target) {
      return new fn(...preArgs, ...args);
    }
    return fn.myCall(thisArg, ...preArgs, ...args);
  }
  // 保持原函数的 prototype（使 instanceof 正常工作）
  bound.prototype = Object.create(fn.prototype);
  return bound;
};
```

```js
function parseQuery(url) {
  const searchStr = url.includes('?') ? url.split('?')[1].split('#')[0] : url;
  const params = new URLSearchParams(searchStr);
  const result = {};
  for (const [key, value] of params) {

      result[key] = [].concat(result[key], value);  // 同名参数转数组
    } else {
      result[key] = value;
    }
  }

}

function parseQueryManual(url) {
  const query = url.split('?')[1]?.split('#')[0];
  if (!query) return {};

  return query.split('&').reduce((acc, pair) => {
    const eqIndex = pair.indexOf('=');
    if (eqIndex === -1) return acc;
    const key = decodeURIComponent(pair.slice(0, eqIndex));
    const value = decodeURIComponent(pair.slice(eqIndex + 1));
    if (acc[key] !== undefined) {
      acc[key] = [].concat(acc[key], value);
    } else {
      acc[key] = value;
    }
    return acc;
  }, {});
}
```

## 15. 实现一个简易 Vue 响应式

**难度**：⭐⭐⭐⭐⭐ | **频率**：🔥🔥🔥🔥

**思路：** Proxy 拦截读写 + 依赖收集（track）+ 派发更新（trigger）

```js
// 当前正在运行的 effect
let activeEffect = null;
// 依赖映射：target → key → Set<effect>
const targetMap = new WeakMap();

function track(target, key) {
  if (!activeEffect) return;
  let depsMap = targetMap.get(target);
  if (!depsMap) targetMap.set(target, depsMap = new Map());
  let deps = depsMap.get(key);
  if (!deps) depsMap.set(key, deps = new Set());
  deps.add(activeEffect);
}

function trigger(target, key) {
  const depsMap = targetMap.get(target);
  if (!depsMap) return;
  const effects = depsMap.get(key);
  if (effects) effects.forEach(effect => effect());
}

function reactive(target) {
  return new Proxy(target, {
    get(target, key, receiver) {
      const result = Reflect.get(target, key, receiver);
      track(target, key);
      return typeof result === 'object' && result !== null
        ? reactive(result)  // 递归响应式
        : result;
    },
    set(target, key, value, receiver) {
      const result = Reflect.set(target, key, value, receiver);
      trigger(target, key);
      return result;
    },
    deleteProperty(target, key) {
      const result = Reflect.deleteProperty(target, key);
      trigger(target, key);
      return result;
    }
  });
}

// watchEffect：自动收集依赖
function watchEffect(fn) {
  function effect() {
    activeEffect = effect;
    fn();
    activeEffect = null;
  }
  effect();  // 立即执行，触发 get 收集依赖
  return () => {
    // 停止监听（简化版，未实现 cleanup）
    targetMap.forEach(depsMap => depsMap.forEach(deps => deps.delete(effect)));
  };
}

// computed
function computed(getter) {
  let cached;
  let dirty = true;

  function effect() {
    dirty = true;
    trigger(computed, 'value');
  }

  return new Proxy({}, {
    get(_, key) {
      if (key === 'value') {
        if (dirty) {
          activeEffect = effect;
          cached = getter();
          activeEffect = null;
          dirty = false;
        }
        track(computed, 'value');
        return cached;
      }
    }
  });
}

  console.log(`count is: ${state.count}`);
});
```

```js
// 方法1：递归
function flat(arr, depth = 1) {
  if (depth === 0) return arr.slice();
  return arr.reduce((result, item) => {
    if (Array.isArray(item)) {
      result.push(...flat(item, depth - 1));
    } else {
      result.push(item);
    }
    return result;
  }, []);
}

// 方法2：迭代（栈）
function flatIterative(arr) {
  const result = [];
  const stack = [...arr];
  while (stack.length) {
    const item = stack.shift();
    if (Array.isArray(item)) {
      stack.unshift(...item);  // 展开放到队首
    } else {
      result.push(item);
    }
  }
  return result;
}
```

```js
class LazyImageLoader {
  #observer;

  constructor(options = {}) {
    this.#observer = new IntersectionObserver(
      (entries) => {
        entries.forEach(entry => {
          if (entry.isIntersecting) {
            this.#loadImage(entry.target);
            this.#observer.unobserve(entry.target);
          }
        });
      },
      {
        rootMargin: options.rootMargin || '100px',  // 提前 100px 加载
        threshold: options.threshold || 0,
      }
    );
  }

  #loadImage(img) {
    const src = img.dataset.src;
    const srcset = img.dataset.srcset;
    if (!src) return;

    // 预加载图片
    const tempImg = new Image();
    tempImg.onload = () => {
      img.src = src;
      if (srcset) img.srcset = srcset;
      img.removeAttribute('data-src');
      img.removeAttribute('data-srcset');
      img.classList.add('loaded');
    };
    tempImg.src = src;
  }

  observe(imgEl) { this.#observer.observe(imgEl); }

  observeAll(selector = '[data-src]') {
    document.querySelectorAll(selector).forEach(img => this.observe(img));
  }

  disconnect() { this.#observer.disconnect(); }
}
```
