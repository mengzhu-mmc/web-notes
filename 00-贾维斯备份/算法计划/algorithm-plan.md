### 滑动窗口
```javascript
function slidingWindow(s) {
  let left = 0, res = 0;
  const map = new Map();
  for (let right = 0; right < s.length; right++) {
    // 扩大窗口，更新状态
    while (/* 窗口需要收缩 */) {
      // 收缩窗口，更新状态
      left++;
    }
    res = Math.max(res, right - left + 1);
  }
  return res;
}
```

### 回溯
```javascript
function backtrack(path, choices) {
  if (/* 满足结束条件 */) {
    result.push([...path]);
    return;
  }
  for (const choice of choices) {
    if (/* 剪枝条件 */) continue;
    path.push(choice);      // 选择
    backtrack(path, ...);    // 递归
    path.pop();              // 撤销
  }
}
```

### BFS
```javascript
function bfs(start) {
  const queue = [start];
  const visited = new Set([start]);
  let level = 0;
  while (queue.length) {
    const size = queue.length;
    for (let i = 0; i < size; i++) {
      const node = queue.shift();
      // 处理当前节点
      for (const next of neighbors(node)) {
        if (!visited.has(next)) {
          visited.add(next);
          queue.push(next);
        }
      }
    }
    level++;
  }
}
```
