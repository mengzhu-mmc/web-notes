## 一、双指针

双指针是前端面试中最常考的算法技巧之一，核心思想是用两个指针在数组或链表上协同移动，避免暴力枚举。

### 1.1 对撞指针（左右指针）

两个指针从数组两端向中间移动，适合有序数组上的搜索问题。

```javascript
// 经典题：两数之和（有序数组版）
function twoSum(nums, target) {
  let left = 0, right = nums.length - 1;
  while (left < right) {
    const sum = nums[left] + nums[right];
    if (sum === target) return [left, right];
    else if (sum < target) left++;
    else right--;
  }
  return [];
}

// 经典题：三数之和（LeetCode 15）
// 思路：排序 + 固定一个数 + 对撞指针找另外两个
function threeSum(nums) {
  nums.sort((a, b) => a - b);
  const result = [];
  for (let i = 0; i < nums.length - 2; i++) {
    if (i > 0 && nums[i] === nums[i - 1]) continue; // 去重
    let left = i + 1, right = nums.length - 1;
    while (left < right) {
      const sum = nums[i] + nums[left] + nums[right];
      if (sum === 0) {
        result.push([nums[i], nums[left], nums[right]]);
        while (left < right && nums[left] === nums[left + 1]) left++;   // 去重
        while (left < right && nums[right] === nums[right - 1]) right--; // 去重
        left++;
        right--;
      } else if (sum < 0) left++;
      else right--;
    }
  }
  return result;
}
```

### 1.2 快慢指针

两个指针以不同速度移动，常用于链表环检测、找中间节点。

```javascript
// 判断链表是否有环（LeetCode 141）
function hasCycle(head) {
  let slow = head, fast = head;
  while (fast && fast.next) {
    slow = slow.next;
    fast = fast.next.next;
    if (slow === fast) return true;
  }
  return false;
}

// 找链表中间节点（LeetCode 876）
function middleNode(head) {
  let slow = head, fast = head;
  while (fast && fast.next) {
    slow = slow.next;
    fast = fast.next.next;
  }
  return slow; // slow 就是中间节点
}
```

> [!tip] 双指针适用场景
> 有序数组求和/查找用对撞指针，链表问题用快慢指针，子串/子数组问题用滑动窗口（也是双指针的变体）。

## 二、滑动窗口

滑动窗口本质是双指针的一种特殊形式，维护一个可变大小的"窗口"在数组/字符串上滑动，适合求解子串、子数组相关问题。

### 模板

```javascript
function slidingWindow(s) {

  while (right < s.length) {

    // 判断是否需要收缩窗口

      left++;
      // 更新窗口内数据...
    }

    // 在这里更新结果
  }
  return result;

```

### 经典题

```javascript
// 无重复字符的最长子串（LeetCode 3）
function lengthOfLongestSubstring(s) {
  const set = new Set();
  let left = 0, maxLen = 0;
  for (let right = 0; right < s.length; right++) {
    while (set.has(s[right])) {
      set.delete(s[left]);
      left++;
    }
    set.add(s[right]);
    maxLen = Math.max(maxLen, right - left + 1);
  }
  return maxLen;
}

// 找到字符串中所有字母异位词（LeetCode 438）
function findAnagrams(s, p) {
  const result = [];
  const need = new Array(26).fill(0);
  const window = new Array(26).fill(0);
  const base = 'a'.charCodeAt(0);

  for (const c of p) need[c.charCodeAt(0) - base]++;

  let left = 0;
  for (let right = 0; right < s.length; right++) {
    window[s[right].charCodeAt(0) - base]++;
    if (right - left + 1 > p.length) {
      window[s[left].charCodeAt(0) - base]--;
      left++;
    }
    if (right - left + 1 === p.length && window.join() === need.join()) {
      result.push(left);
    }
  }
  return result;
}
```

## 三、BFS（广度优先搜索）

BFS 用队列实现，按层遍历，适合求最短路径、层序遍历等问题。

### 模板

```javascript
function bfs(start) {
  const queue = [start];
  const visited = new Set([start]);
  let step = 0;

  while (queue.length > 0) {
    const size = queue.length;
    for (let i = 0; i < size; i++) {
      const node = queue.shift();
      // 判断是否到达终点
      // 将相邻节点加入队列
      for (const neighbor of getNeighbors(node)) {
        if (!visited.has(neighbor)) {
          visited.add(neighbor);
          queue.push(neighbor);
        }
      }
    }
    step++;
  }
}
```

### 经典题

```javascript
// 二叉树的层序遍历（LeetCode 102）
function levelOrder(root) {
  if (!root) return [];
  const result = [];
  const queue = [root];
  while (queue.length > 0) {
    const level = [];
    const size = queue.length;
    for (let i = 0; i < size; i++) {
      const node = queue.shift();
      level.push(node.val);
      if (node.left) queue.push(node.left);
      if (node.right) queue.push(node.right);
    }
    result.push(level);
  }
  return result;
}

// 岛屿数量（LeetCode 200）—— BFS 版
function numIslands(grid) {
  let count = 0;
  const rows = grid.length, cols = grid[0].length;
  const dirs = [[0, 1], [0, -1], [1, 0], [-1, 0]];

  for (let i = 0; i < rows; i++) {
    for (let j = 0; j < cols; j++) {
      if (grid[i][j] === '1') {
        count++;
        const queue = [[i, j]];
        grid[i][j] = '0'; // 标记已访问
        while (queue.length) {
          const [x, y] = queue.shift();
          for (const [dx, dy] of dirs) {
            const nx = x + dx, ny = y + dy;
            if (nx >= 0 && nx < rows && ny >= 0 && ny < cols && grid[nx][ny] === '1') {
              grid[nx][ny] = '0';
              queue.push([nx, ny]);
            }
          }
        }
      }
    }
  }
  return count;
}
```

## 四、DFS（深度优先搜索）与回溯

DFS 用递归或栈实现，适合排列组合、路径搜索、树的遍历等问题。回溯是 DFS 的一种应用模式，核心是"选择 → 递归 → 撤销选择"。

### 回溯模板

```javascript
function backtrack(path, choices) {
  if (/* 满足结束条件 */) {
    result.push([...path]); // 注意要拷贝
    return;
  }
  for (const choice of choices) {
    // 做选择
    path.push(choice);
    // 递归
    backtrack(path, /* 更新后的选择列表 */);
    // 撤销选择
    path.pop();
  }
}
```

### 经典题

```javascript
// 全排列（LeetCode 46）
function permute(nums) {
  const result = [];
  const used = new Array(nums.length).fill(false);

  function backtrack(path) {
    if (path.length === nums.length) {
      result.push([...path]);
      return;
    }
    for (let i = 0; i < nums.length; i++) {
      if (used[i]) continue;
      used[i] = true;
      path.push(nums[i]);
      backtrack(path);
      path.pop();
      used[i] = false;
    }
  }

  backtrack([]);
  return result;
}

// 子集（LeetCode 78）
function subsets(nums) {
  const result = [];
  function backtrack(start, path) {
    result.push([...path]);
    for (let i = start; i < nums.length; i++) {
      path.push(nums[i]);
      backtrack(i + 1, path);
      path.pop();
    }
  }
  backtrack(0, []);
  return result;
}

// 括号生成（LeetCode 22）
function generateParenthesis(n) {
  const result = [];
  function backtrack(str, open, close) {
    if (str.length === 2 * n) {
      result.push(str);
      return;
    }
    if (open < n) backtrack(str + '(', open + 1, close);
    if (close < open) backtrack(str + ')', open, close + 1);
  }
  backtrack('', 0, 0);
  return result;
}
```

## 五、动态规划

动态规划的核心是把大问题拆成子问题，通过记录子问题的解来避免重复计算。关键步骤：定义状态 → 写出状态转移方程 → 确定初始值和遍历顺序。

### 5.1 一维 DP

```javascript
// 爬楼梯（LeetCode 70）
// dp[i] = dp[i-1] + dp[i-2]
function climbStairs(n) {
  if (n <= 2) return n;
  let prev2 = 1, prev1 = 2;
  for (let i = 3; i <= n; i++) {
    const curr = prev1 + prev2;
    prev2 = prev1;
    prev1 = curr;
  }
  return prev1;
}

// 最大子数组和（LeetCode 53）
// dp[i] = max(nums[i], dp[i-1] + nums[i])
function maxSubArray(nums) {
  let maxSum = nums[0], currSum = nums[0];
  for (let i = 1; i < nums.length; i++) {
    currSum = Math.max(nums[i], currSum + nums[i]);
    maxSum = Math.max(maxSum, currSum);
  }
  return maxSum;
}
```

### 5.2 二维 DP

```javascript
// 最长公共子序列（LeetCode 1143）
function longestCommonSubsequence(text1, text2) {
  const m = text1.length, n = text2.length;
  const dp = Array.from({ length: m + 1 }, () => new Array(n + 1).fill(0));

  for (let i = 1; i <= m; i++) {
    for (let j = 1; j <= n; j++) {
      if (text1[i - 1] === text2[j - 1]) {
        dp[i][j] = dp[i - 1][j - 1] + 1;
      } else {
        dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
      }
    }
  }
  return dp[m][n];
}
```

### 5.3 买卖股票系列

```javascript
// 买卖股票的最佳时机（LeetCode 121）—— 只能买卖一次
function maxProfit(prices) {
  let minPrice = Infinity, maxProfit = 0;
  for (const price of prices) {
    minPrice = Math.min(minPrice, price);
    maxProfit = Math.max(maxProfit, price - minPrice);
  }
  return maxProfit;
}
```

> [!tip] DP 解题思路
> 先想暴力递归怎么写 → 加备忘录变成记忆化搜索 → 改写成自底向上的 DP 表。大多数 DP 题的状态转移方程都可以从"最后一步怎么做"来推导。

## 六、常用数据结构技巧

### 6.1 哈希表（Map/Set）

```javascript
// 两数之和（LeetCode 1）—— 最经典的哈希表应用
function twoSum(nums, target) {
  const map = new Map();
  for (let i = 0; i < nums.length; i++) {
    const complement = target - nums[i];
    if (map.has(complement)) return [map.get(complement), i];
    map.set(nums[i], i);
  }
}

// 有效的字母异位词（LeetCode 242）
function isAnagram(s, t) {
  if (s.length !== t.length) return false;
  const count = new Array(26).fill(0);
  const base = 'a'.charCodeAt(0);
  for (let i = 0; i < s.length; i++) {
    count[s.charCodeAt(i) - base]++;
    count[t.charCodeAt(i) - base]--;
  }
  return count.every(c => c === 0);
}
```

### 6.2 栈

```javascript
// 有效的括号（LeetCode 20）
function isValid(s) {
  const stack = [];
  const map = { ')': '(', ']': '[', '}': '{' };
  for (const c of s) {
    if (map[c]) {
      if (stack.pop() !== map[c]) return false;
    } else {
      stack.push(c);
    }
  }
  return stack.length === 0;
}

// 字符串解码（LeetCode 394）
function decodeString(s) {
  const numStack = [], strStack = [];
  let num = 0, str = '';
  for (const c of s) {
    if (c >= '0' && c <= '9') {
      num = num * 10 + Number(c);
    } else if (c === '[') {
      numStack.push(num);
      strStack.push(str);
      num = 0;
      str = '';
    } else if (c === ']') {
      str = strStack.pop() + str.repeat(numStack.pop());
    } else {
      str += c;
    }
  }
  return str;
}
```
