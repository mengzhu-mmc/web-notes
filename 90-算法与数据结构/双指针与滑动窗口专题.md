## 一、双指针技巧

### 1.1 双指针的分类

| 类型 | 适用场景 | 典型题目 |
|------|----------|----------|
| **对撞指针** | 有序数组，从两端向中间移动 | 两数之和、盛最多水的容器 |
| **快慢指针** | 链表、数组去重、判断环 | 反转链表、删除重复项 |
| **滑动窗口** | 子串/子数组问题 | 最长无重复子串、最小覆盖子串 |
| **分离指针** | 两个数组/链表同时遍历 | 合并有序数组、相交链表 |

### 1.2 对撞指针模板

```javascript
function twoPointer(arr, target) {
  let left = 0;
  let right = arr.length - 1;
  
  while (left < right) {
    const sum = arr[left] + arr[right];
    
    if (sum === target) {
      return [left, right];
    } else if (sum < target) {
      left++;  // 需要更大的值
    } else {
      right--; // 需要更小的值
    }
  }
  
  return null;
}
```

## 二、经典双指针题目

### 2.1 两数之和 II（有序数组）

**题目**：给定已按升序排列的数组，找出和为目标值的两个数。

```javascript
function twoSum(numbers, target) {
  let left = 0;
  let right = numbers.length - 1;
  
  while (left < right) {
    const sum = numbers[left] + numbers[right];
    
    if (sum === target) {
      return [left + 1, right + 1]; // 返回 1-based 索引
    } else if (sum < target) {
      left++;
    } else {
      right--;
    }
  }
  
  return [];
}

// 时间复杂度：O(n)
// 空间复杂度：O(1)
```

### 2.2 三数之和

**题目**：找出数组中所有和为 0 的三元组，且不重复。

```javascript
function threeSum(nums) {
  const result = [];
  nums.sort((a, b) => a - b); // 先排序
  
  for (let i = 0; i < nums.length - 2; i++) {
    // 跳过重复元素
    if (i > 0 && nums[i] === nums[i - 1]) continue;
    
    let left = i + 1;
    let right = nums.length - 1;
    
    while (left < right) {
      const sum = nums[i] + nums[left] + nums[right];
      
      if (sum === 0) {
        result.push([nums[i], nums[left], nums[right]]);
        
        // 跳过重复元素
        while (left < right && nums[left] === nums[left + 1]) left++;
        while (left < right && nums[right] === nums[right - 1]) right--;
        
        left++;
        right--;
      } else if (sum < 0) {
        left++;
      } else {
        right--;
      }
    }
  }
  
  return result;
}

// 时间复杂度：O(n²)
// 空间复杂度：O(1) 或 O(log n)（排序）
```

### 2.3 盛最多水的容器

**题目**：给定 n 条垂线，找出两条线使得容器可以盛最多的水。

```javascript
function maxArea(height) {
  let left = 0;
  let right = height.length - 1;
  let maxWater = 0;
  
  while (left < right) {
    // 面积 = 宽度 * 高度（取较短的线）
    const width = right - left;
    const h = Math.min(height[left], height[right]);
    maxWater = Math.max(maxWater, width * h);
    
    // 移动较短的线（因为移动长的线只会使面积更小）
    if (height[left] < height[right]) {
      left++;
    } else {
      right--;
    }
  }
  
  return maxWater;
}

// 核心思想：移动较短的线才有可能获得更大的面积
// 时间复杂度：O(n)
// 空间复杂度：O(1)
```

### 2.4 接雨水

**题目**：给定 n 个非负整数表示每个宽度为 1 的柱子的高度，计算按此排列的柱子下雨后能接多少雨水。

```javascript
function trap(height) {
  let left = 0;
  let right = height.length - 1;
  let leftMax = 0;
  let rightMax = 0;
  let water = 0;
  
  while (left < right) {
    if (height[left] < height[right]) {
      // 左边是瓶颈
      if (height[left] >= leftMax) {
        leftMax = height[left];
      } else {
        water += leftMax - height[left];
      }
      left++;
    } else {
      // 右边是瓶颈
      if (height[right] >= rightMax) {
        rightMax = height[right];
      } else {
        water += rightMax - height[right];
      }
      right--;
    }
  }
  
  return water;
}

// 核心思想：每个位置能接的雨水 = min(左边最高, 右边最高) - 当前高度
// 时间复杂度：O(n)
// 空间复杂度：O(1)
```

## 三、滑动窗口技巧

### 3.1 滑动窗口模板

```javascript
function slidingWindow(s) {
  const window = new Map(); // 或 Set，根据题目需求
  let left = 0;
  let right = 0;
  let result = ...;
  
  while (right < s.length) {
    // 扩大窗口：将 s[right] 加入窗口
    const c = s[right];
    window.set(c, (window.get(c) || 0) + 1);
    right++;
    
    // 收缩窗口（当窗口不满足条件时）
    while (window needs shrink) {
      const d = s[left];
      window.set(d, window.get(d) - 1);
      if (window.get(d) === 0) window.delete(d);
      left++;
    }
    
    // 更新结果
    result = Math.max(result, right - left);
  }
  
  return result;
}
```

### 3.2 滑动窗口的分类

| 类型 | 特征 | 典型题目 |
|------|------|----------|
| **固定窗口** | 窗口大小固定 | 滑动窗口最大值、找到字符串中所有字母异位词 |
| **可变窗口** | 窗口大小动态变化 | 最长无重复子串、最小覆盖子串 |
| **多指针窗口** | 需要多个窗口或复杂条件 | 串联所有单词的子串 |

## 四、经典滑动窗口题目

### 4.1 最长无重复子串（可变窗口）

**题目**：给定一个字符串，找出不含有重复字符的最长子串的长度。

```javascript
function lengthOfLongestSubstring(s) {
  const charSet = new Set();
  let left = 0;
  let maxLen = 0;
  
  for (let right = 0; right < s.length; right++) {
    // 如果有重复，收缩窗口
    while (charSet.has(s[right])) {
      charSet.delete(s[left]);
      left++;
    }
    
    // 扩大窗口
    charSet.add(s[right]);
    maxLen = Math.max(maxLen, right - left + 1);
  }
  
  return maxLen;
}

// 时间复杂度：O(n)
// 空间复杂度：O(min(m, n))，m 是字符集大小
```

### 4.2 最小覆盖子串（可变窗口）

**题目**：给定字符串 S 和 T，在 S 中找到包含 T 所有字符的最小子串。

```javascript
function minWindow(s, t) {
  const need = new Map(); // T 中字符的需求
  const window = new Map(); // 当前窗口中的字符
  
  // 初始化 need
  for (const c of t) {
    need.set(c, (need.get(c) || 0) + 1);
  }
  
  let left = 0;
  let right = 0;
  let valid = 0; // 满足条件的字符数
  let start = 0;
  let len = Infinity;
  
  while (right < s.length) {
    const c = s[right];
    right++;
    
    // 扩大窗口
    if (need.has(c)) {
      window.set(c, (window.get(c) || 0) + 1);
      if (window.get(c) === need.get(c)) {
        valid++;
      }
    }
    
    // 收缩窗口（当窗口已包含所有所需字符）
    while (valid === need.size) {
      // 更新最小窗口
      if (right - left < len) {
        start = left;
        len = right - left;
      }
      
      const d = s[left];
      left++;
      
      if (need.has(d)) {
        if (window.get(d) === need.get(d)) {
          valid--;
        }
        window.set(d, window.get(d) - 1);
      }
    }
  }
  
  return len === Infinity ? "" : s.substring(start, start + len);
}

// 时间复杂度：O(n)
// 空间复杂度：O(m)，m 是字符集大小
```

### 4.3 滑动窗口最大值（固定窗口）

**题目**：给定数组和窗口大小 k，返回每个滑动窗口的最大值。

```javascript
function maxSlidingWindow(nums, k) {
  const result = [];
  const deque = []; // 双端队列，存储索引，保持递减
  
  for (let i = 0; i < nums.length; i++) {
    // 移除窗口外的元素
    while (deque.length && deque[0] <= i - k) {
      deque.shift();
    }
    
    // 保持队列递减（移除所有小于当前元素的）
    while (deque.length && nums[deque[deque.length - 1]] < nums[i]) {
      deque.pop();
    }
    
    deque.push(i);
    
    // 记录结果
    if (i >= k - 1) {
      result.push(nums[deque[0]]);
    }
  }
  
  return result;
}

// 核心思想：单调队列
// 时间复杂度：O(n)
// 空间复杂度：O(k)
```

### 4.4 找到字符串中所有字母异位词（固定窗口）

**题目**：给定字符串 s 和 p，找到 s 中所有 p 的异位词的子串，返回起始索引。

```javascript
function findAnagrams(s, p) {
  const result = [];
  const need = new Map();
  const window = new Map();
  
  // 初始化 need
  for (const c of p) {
    need.set(c, (need.get(c) || 0) + 1);
  }
  
  let left = 0;
  let right = 0;
  let valid = 0;
  
  while (right < s.length) {
    const c = s[right];
    right++;
    
    if (need.has(c)) {
      window.set(c, (window.get(c) || 0) + 1);
      if (window.get(c) === need.get(c)) {
        valid++;
      }
    }
    
    // 窗口大小固定为 p.length
    while (right - left >= p.length) {
      if (valid === need.size) {
        result.push(left);
      }
      
      const d = s[left];
      left++;
      
      if (need.has(d)) {
        if (window.get(d) === need.get(d)) {
          valid--;
        }
        window.set(d, window.get(d) - 1);
      }
    }
  }
  
  return result;
}

// 时间复杂度：O(n)
// 空间复杂度：O(m)，m 是字符集大小
```

## 五、快慢指针专题

### 5.1 判断链表是否有环

```javascript
function hasCycle(head) {
  if (!head || !head.next) return false;
  
  let slow = head;
  let fast = head.next;
  
  while (fast && fast.next) {
    if (slow === fast) return true;
    slow = slow.next;
    fast = fast.next.next;
  }
  
  return false;
}

// 时间复杂度：O(n)
// 空间复杂度：O(1)
```

### 5.2 删除有序数组中的重复项

```javascript
function removeDuplicates(nums) {
  if (nums.length === 0) return 0;
  
  let slow = 0;
  
  for (let fast = 1; fast < nums.length; fast++) {
    if (nums[fast] !== nums[slow]) {
      slow++;
      nums[slow] = nums[fast];
    }
  }
  
  return slow + 1;
}

// 时间复杂度：O(n)
// 空间复杂度：O(1)
```

## 六、解题技巧总结

### 6.1 何时使用双指针

- ✅ 有序数组的查找、求和
- ✅ 需要原地修改数组
- ✅ 链表相关问题
- ✅ 判断回文、反转

### 6.2 何时使用滑动窗口

- ✅ 子串/子数组问题
- ✅ 需要满足某些条件的连续序列
- ✅ 涉及频率统计的字符串问题

### 6.3 常见错误

1. **边界条件**：注意 `left < right` vs `left <= right`
2. **重复元素处理**：三数之和等需要去重
3. **窗口收缩时机**：确保在满足条件时才收缩
4. **索引越界**：检查 `right` 和 `left` 的边界
